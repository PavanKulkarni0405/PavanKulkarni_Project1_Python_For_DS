# Define a function to clean and convert the values
def convert_value(val):
    if isinstance(val, str):
        if val.endswith('M'):
            return float(val.replace('€', '').replace('M', '')) * 1000000
        elif val.endswith('K'):
            return float(val.replace('€', '').replace('K', '')) * 1000
        else:
            return float(val.replace('€', ''))
    else:
        return val
